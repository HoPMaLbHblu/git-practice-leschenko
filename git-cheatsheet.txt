git init - Создает новый локальный репозиторий и/или инициализируется в него
git status - Показывает состояние файлов в репозитории
git add <файл> - Добавляет файлы в отслеживание гитом
git commit -m "текст" - Сохраняет изменения с комментарием
git log - Показывает историю коммитов подробно
git log --oneline - Показывает историю коммитов сокращенно
git push - Отправляет коммиты в удаленный репозиторий
git switch <ветка> - Переключается на указанную ветку
git remote add origin <url> - Связывает локальный репозиторий с удаленным
git branch <ветка> - Создает новую ветку
git pull - Загружает изменения из удаленного репозитория
cd - Переход по директориям
touch - Создание файла
cat - Объединение файлов
